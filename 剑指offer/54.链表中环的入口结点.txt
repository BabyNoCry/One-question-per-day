题目描述
给一个链表，若其中包含环，请找出该链表的环的入口结点，否则，输出null。


/*
struct ListNode {
    int val;
    struct ListNode *next;
    ListNode(int x) :
        val(x), next(NULL) {
    }
};
*/
class Solution {
public:
    
    ListNode* EntryNodeOfLoop(ListNode* pHead)
    {
        if(pHead == nullptr)
            return NULL;
        ListNode* slow = pHead;
        ListNode* fast = pHead;
        while(fast && fast ->next)
        {
            slow = slow ->next;
            fast = fast->next->next;
            if(slow == fast)
            {
                fast = pHead;
                while(slow -fast)
                {
                    fast = fast->next;
                    slow = slow ->next;
                }
                return slow;
            }
        }
        
        return NULL;

    }
};
